=========================================================================
5) SHARED AnchorStatus COMPONENT (keeps every surface consistent)
=========================================================================
One component, two modes, driven by an anchorStatus field:
- mode="status" (operational surfaces: settled Approved Record card, auto-handled receipt,
  proposal detail): shield + short label + one-line guarantee + inline "Verify" + optional
  "View full record" link. No hash block.
- mode="proof" (canonical audit record, rule audit trail): the above PLUS the mono
  merkle/tx/block block and the prominent "Verify on-chain" button.
States (honest, batched-anchoring progression):
    "pending_next_batch" → clock icon, "Recorded · anchoring in next batch", Verify DISABLED,
       hashes shown as "pending next batch".
    "anchored" → shield-check (success tint), "Anchored · tamper-evident", Verify ENABLED,
       real (truncated) hashes.
Never claim "anchored" for a record still pending its batch. Verify is disabled until
anchored.

DATA MODEL (extend existing types)
=========================================================================
type AuditEventType = "approved" | "auto_approved" | "rule_change" | "trust_granted"
  | "trust_revoked" | "flagged";
type AnchorStatus = "pending_next_batch" | "anchored";

interface AnchorProof {
  status: AnchorStatus;
  auditId: string;
  merkleRoot?: string;   // present when anchored
  baseTx?: string;       // present when anchored
  block?: number;
  anchoredAtLabel?: string;
  verifyHref?: string;
}
interface LifecycleStep { label: string; timestamp: string; note?: string;
  kind: "ok" | "alert"; }
interface AuditRecord {
  id: string;              // "AUD-7K2M"
  eventType: AuditEventType;
  summary: string;         // plain-language
  counterparty?: string;
  amount?: number;
  actor: string;           // "sarah@meridian" | "system"
  occurredAtLabel: string;
  lifecycle: LifecycleStep[];
  linked: { kind: "vendor"|"proposal"|"rule"; label: string; refId: string }[];
  anchor: AnchorProof;
  // link back to an operational item, if this record is an approved/executed payment:
  proposalId?: string;
}
